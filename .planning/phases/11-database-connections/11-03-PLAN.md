   ```python
   def import_to_duckdb(self, query: str, duckdb_path: str, table_name: str) -> str:
       """Execute query and import results to DuckDB.
       
       Returns:
           Table name in DuckDB where data was imported
       """
       df = self.execute_query_df(query)
       conn = duckdb.connect(duckdb_path)
       conn.execute(f"CREATE TABLE IF NOT EXISTS {table_name} AS SELECT * FROM df")
       conn.close()
       return table_name
   ```

   ```python
   class DatabaseSchemaDiscoverer:
       def list_tables(self) -> List[str]:
           """Return list of table names."""
           raise NotImplementedError
       
       def describe_table(self, table_name: str) -> dict:
           """Return table structure: columns, types, nullable."""
           raise NotImplementedError
   ```

2. **Implement SQLSchemaDiscoverer (MySQL, PostgreSQL, SQLite):**
   ```python
   class SQLSchemaDiscoverer(DatabaseSchemaDiscoverer):
       def __init__(self, engine: Engine):
           self.engine = engine
       
       def list_tables(self) -> List[str]:
           insp = inspect(self.engine)
           return insp.get_table_names()
       
       def describe_table(self, table_name: str) -> dict:
           insp = inspect(self.engine)
           columns = insp.get_columns(table_name)
           return {
               'table': table_name,
               'columns': [{'name': c['name'], 'type': str(c['type']), 'nullable': c['nullable']} for c in columns]
           }
   ```

3. **Implement MongoDBSchemaDiscoverer:**
   ```python
   class MongoDBSchemaDiscoverer(DatabaseSchemaDiscoverer):
       def __init__(self, client: MongoClient, db_name: str):
           self.db = client[db_name]
       
       def list_tables(self) -> List[str]:
           return self.db.list_collection_names()
       
       def describe_table(self, collection_name: str) -> dict:
           # Sample documents to infer schema
           sample = self.db[collection_name].find().limit(10)
           fields = set()
           for doc in sample:
               fields.update(doc.keys())
           return {
               'collection': collection_name,
               'fields': list(fields),
               'type': 'document'
           }
   ```

4. **Output format:**
   - Markdown table for terminal display
   - JSON option for programmatic use

**Do NOT:**
- Query full table contents (sample only)
- Make network calls for schema (use database metadata)
- Include internal system tables by default

   ```
   db-cli describe-table <profile> <table> [--output table|json]
   db-cli import <profile> <query> [--table-name NAME] [--duckdb PATH]
   ```

2. **Profile selection:**
   - Load `db_connections.json` (auto-discover)
   - Use profile specified by `--profile` flag
   - Error if profile not found

3. **Query command implementation:**
   ```python
   def db_query_cmd(args):
       config = load_db_config()
       profile = config[args.profile]
       connector = create_connector(profile)
       
       if args.file:
           query = open(args.file).read()
       else:
           query = args.query
       
       if args.output == 'json':
           df = connector.execute_query_df(query)
           print(df.to_json(orient='records'))
       else:
           df = connector.execute_query_df(query)
           print(df.to_markdown())
   ```

4. **Import command:**
   ```python
   def db_import_cmd(args):
       config = load_db_config()
       profile = config[args.profile]
       connector = create_connector(profile)
       
       table_name = args.table_name or generate_table_name(args.query)
       duckdb_path = args.duckdb or 'workspace.duckdb'
       
       result_table = connector.import_to_duckdb(args.query, duckdb_path, table_name)
       print(f"Imported to DuckDB table: {result_table}")
   ```

5. **Output formats:**
   - `table`: Markdown table (default)
   - `json`: JSON array

6. **Timeout handling:**
   - Default 30 seconds
   - `--timeout` flag to override
   - Timeout exception with clear message

- Use interactive prompts (CLI should be non-interactive)
- Auto-import without explicit `import` command
