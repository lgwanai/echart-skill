```python
class ServerLifecycle:
    """Manage server process lifecycle with PID tracking."""

    def __init__(self, port: int):
        self.port = port
        self.pid_file = PID_DIR / f"server_{port}.pid"
        self.last_request_file = PID_DIR / f"server_{port}_last_request"
        PID_DIR.mkdir(parents=True, exist_ok=True)

    def write_pid(self) -> None: ...
    def read_pid(self) -> int | None: ...
    def clear_pid(self) -> None: ...
    def update_last_request(self) -> None: ...
    def is_server_active(self, pid: int) -> bool: ...
    def kill_orphan(self) -> bool: ...
    def should_shutdown(self) -> bool: ...
```

Key implementation:
- write_pid(): Write os.getpid() to file
- read_pid(): Read and parse PID from file
- clear_pid(): Remove PID and last_request files
- is_server_active(): Use os.kill(pid, 0) to check process
- kill_orphan(): Check if PID file exists but process dead
- should_shutdown(): Compare last_request with current time

Note: Ensure PID_DIR is created with mkdir(parents=True, exist_ok=True).
